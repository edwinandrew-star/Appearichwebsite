import bpy import random # Clear existing meshes bpy.ops.object.select_all(action='DESELECT') bpy.ops.object.select_by_type(type='MESH') bpy.ops.object.delete() # Create a forest background (simple example) def create_forest(): for i in range(10): bpy.ops.mesh.primitive_plane_add(size=10, enter_editmode=False, align='WORLD', location=(random.uniform(-30, 30), random.uniform(-30, 30), 0)) mat = bpy.data.materials.new(name="GrassMaterial") mat.diffuse_color = (0.1, 0.5, 0.1, 1) # Green bpy.context.object.data.materials.append(mat) create_forest() # Create a coconut bpy.ops.mesh.primitive_uv_sphere_add(radius=0.5, enter_editmode=False, location=(0, 0, 5)) coconut = bpy.context.object coconut.name = 'Coconut' coconut_material = bpy.data.materials.new(name="CoconutMaterial") coconut_material.diffuse_color = (0.5, 0.3, 0.2, 1) # Brown coconut.data.materials.append(coconut_material) # Animate coconut falling coconut.location.z = 5 coconut.keyframe_insert(data_path="location", frame=1) coconut.location.z = 0 coconut.keyframe_insert(data_path="location", frame=30) # Create perfume bottle bpy.ops.mesh.primitive_cylinder_add(radius=0.3, depth=1, location=(0, 0, 0)) bottle = bpy.context.object bottle.name = 'PerfumeBottle' bottle_material = bpy.data.materials.new(name="BottleMaterial") bottle_material.diffuse_color = (1, 0.85, 0.8, 1) # Light pink bottle.data.materials.append(bottle_material) # Animate filling the bottle bottle.location.z = 0 bottle.keyframe_insert(data_path="scale", frame=1) bottle.scale.z = 2 bottle.keyframe_insert(data_path="scale", frame=60) # Display perfume details def create_text(text, location, size=1): bpy.ops.object.text_add(location=location) txt = bpy.context.object txt.data.body = text txt.scale = (size, size, size) create_text("Appearich", (0, -2, 1), size=1) create_text("Coconut Passion", (0, -2, 2), size=0.8) # Set camera view bpy.ops.object.camera_add(location=(5, -5, 5)) bpy.context.scene.camera = bpy.context.object bpy.context.object.rotation_euler = (1.1, 0, 0.8) # Add lighting bpy.ops.object.light_add(type='SUN', location=(5, -5, 10)) # Render setup bpy.context.scene.render.filepath = "//perfume_animation" bpy.context.scene.render.image_settings.file_format = 'PNG' bpy.ops.render.render(write_still=True) print("Animation setup done!")
